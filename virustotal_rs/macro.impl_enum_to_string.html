Macro to generate to_string implementations for enums with simple string mappings

This macro creates a to_string method that returns &'static str for enum variants. It reduces code duplication and complexity by generating the match statement automatically.

ยงExamples

use virustotal_rs::impl_enum_to_string;

#[derive(Debug, Clone, Copy)]
enum Status {
    Active,
    Inactive,
}

impl_enum_to_string! {
    Status {
        Active => "active",
        Inactive => "inactive",
    }
}

assert_eq!(Status::Active.to_string(), "active");
