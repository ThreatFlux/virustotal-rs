1//! Upload operations for private files
2//!
3//! This module contains methods for uploading files for private scanning,
4//! including both regular and large file uploads.
5
6use super::{PrivateFilesClient, Result};
7use crate::private_files::{PrivateFileUploadParams, PrivateFileUploadResponse, UploadUrlResponse};

9impl<'a> PrivateFilesClient<'a> {
10    /// Upload a file for private scanning (files up to 32MB)
11    ///
12    /// POST /private/files
13    ///
14    /// The file contents should be provided as bytes.
15    /// For files larger than 32MB, use `upload_large_file` instead.
16    pub async fn upload_file(
17        &self,
18        file_data: &[u8],
19        params: Option<PrivateFileUploadParams>,
20    ) -> Result<PrivateFileUploadResponse> {
21        let mut form = reqwest::multipart::Form::new().part(
22            "file",
23            reqwest::multipart::Part::bytes(file_data.to_vec()).file_name("file"),
24        );
25
26        // Add optional parameters
27        if let Some(p) = params {
28            form = self.add_upload_params_to_form(form, p)?;
29        }
30
31        self.client.post_multipart("private/files", form).await
32    }
33
34    /// Helper method to add upload parameters to multipart form
35    fn add_upload_params_to_form(
36        &self,
37        mut form: reqwest::multipart::Form,
38        params: PrivateFileUploadParams,
39    ) -> Result<reqwest::multipart::Form> {
40        form = self.add_sandbox_params(form, &params);
41        form = self.add_execution_params(form, &params);
42        form = self.add_storage_params(form, &params);
43        form = self.add_interaction_params(form, &params);
44        Ok(form)
45    }
46
47    /// Add sandbox-related parameters to form
48    fn add_sandbox_params(
49        &self,
50        mut form: reqwest::multipart::Form,
51        params: &PrivateFileUploadParams,
52    ) -> reqwest::multipart::Form {
53        if let Some(disable_sandbox) = params.disable_sandbox {
54            form = form.text("disable_sandbox", disable_sandbox.to_string());
55        }
56        if let Some(enable_internet) = params.enable_internet {
57            form = form.text("enable_internet", enable_internet.to_string());
58        }
59        if let Some(intercept_tls) = params.intercept_tls {
60            form = form.text("intercept_tls", intercept_tls.to_string());
61        }
62        form
63    }
64
65    /// Add execution-related parameters to form
66    fn add_execution_params(
67        &self,
68        mut form: reqwest::multipart::Form,
69        params: &PrivateFileUploadParams,
70    ) -> reqwest::multipart::Form {
71        if let Some(command_line) = &params.command_line {
72            form = form.text("command_line", command_line.clone());
73        }
74        if let Some(password) = &params.password {
75            form = form.text("password", password.clone());
76        }
77        form
78    }
79
80    /// Add storage-related parameters to form
81    fn add_storage_params(
82        &self,
83        mut form: reqwest::multipart::Form,
84        params: &PrivateFileUploadParams,
85    ) -> reqwest::multipart::Form {
86        if let Some(retention_period_days) = params.retention_period_days {
87            form = form.text("retention_period_days", retention_period_days.to_string());
88        }
89        if let Some(storage_region) = &params.storage_region {
90            form = form.text("storage_region", storage_region.clone());
91        }
92        form
93    }
94
95    /// Add interaction-related parameters to form
96    fn add_interaction_params(
97        &self,
98        mut form: reqwest::multipart::Form,
99        params: &PrivateFileUploadParams,
100    ) -> reqwest::multipart::Form {
101        if let Some(interaction_sandbox) = &params.interaction_sandbox {
102            form = form.text("interaction_sandbox", interaction_sandbox.clone());
103        }
104        if let Some(interaction_timeout) = params.interaction_timeout {
105            form = form.text("interaction_timeout", interaction_timeout.to_string());
106        }
107        if let Some(locale) = &params.locale {
108            form = form.text("locale", locale.clone());
109        }
110        form
111    }
