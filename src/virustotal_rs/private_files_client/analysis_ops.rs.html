1//! Analysis operations for private files
2//!
3//! This module contains methods for handling private analysis operations
4//! including listing, retrieving, and managing analyses.
5
6use super::{PrivateFilesClient, Result};
7use crate::objects::{Collection, CollectionIterator};
8use crate::private_files::{PrivateAnalysis, PrivateAnalysisResponse};
9use crate::url_utils::EndpointBuilder;
10
11impl<'a> PrivateFilesClient<'a> {
12    /// List private analyses
13    ///
14    /// GET /private/analyses
15    ///
16    /// Returns a list of the last private analyses sorted by most recent first.
17    /// Use order="date-" to reverse the order.
18    pub async fn list_analyses(
19        &self,
20        limit: Option<u32>,
21        cursor: Option<&str>,
22        order: Option<&str>,
23    ) -> Result<Collection<PrivateAnalysis>> {
24        let endpoint = EndpointBuilder::new()
25            .raw_segment("private")
26            .raw_segment("analyses")
27            .query_opt("limit", limit.map(|l| l.min(40))) // Maximum 40 analyses
28            .query_opt("cursor", cursor)
29            .query("order", order.unwrap_or("date-")) // Default to date- (most recent first)
30            .build();
31
32        self.client.get(&endpoint).await
33    }
34
35    /// Get an iterator for listing private analyses
36    pub fn list_analyses_iterator(&self) -> CollectionIterator<'_, PrivateAnalysis> {
37        CollectionIterator::new(self.client, "private/analyses".to_string())
38    }
39
40    /// Get a single private analysis
41    ///
42    /// GET /private/analyses/{id}
43    ///
44    /// Returns information about a specific analysis including file metadata.
45    /// The status can be "queued", "in-progress", or "completed".
46    pub async fn get_single_analysis(&self, analysis_id: &str) -> Result<PrivateAnalysisResponse> {
47        let url = format!("private/analyses/{}", urlencoding::encode(analysis_id));
48        self.client.get(&url).await
49    }
50}
