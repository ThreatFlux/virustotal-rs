1//! Analysis-related types and functionality for private files
2
3use super::types::{AnalysisStats, EngineResult, FileInfo};
4use crate::objects::Object;
5use serde::{Deserialize, Serialize};
6use std::collections::HashMap;
7
8/// Private analysis status
9#[derive(Debug, Clone, Serialize, Deserialize)]
10pub struct PrivateAnalysis {
11    #[serde(flatten)]
12    pub object: Object<PrivateAnalysisAttributes>,
13}
14
15/// Attributes for private analysis
16#[derive(Debug, Clone, Serialize, Deserialize)]
17pub struct PrivateAnalysisAttributes {
18    /// Analysis status (queued, in-progress, completed)
19    #[serde(skip_serializing_if = "Option::is_none")]
20    pub status: Option<String>,
21
22    /// Analysis statistics
23    #[serde(skip_serializing_if = "Option::is_none")]
24    pub stats: Option<AnalysisStats>,
25
26    /// Analysis results by engine
27    #[serde(skip_serializing_if = "Option::is_none")]
28    pub results: Option<HashMap<String, EngineResult>>,
29
30    /// Analysis date
31    #[serde(skip_serializing_if = "Option::is_none")]
32    pub date: Option<i64>,
33
34    /// Additional attributes
35    #[serde(flatten)]
36    pub additional_attributes: HashMap<String, serde_json::Value>,
37}
38
39/// Response for getting a single private analysis with file info
40#[derive(Debug, Clone, Serialize, Deserialize)]
41pub struct PrivateAnalysisResponse {
42    pub data: PrivateAnalysis,
43
44    #[serde(skip_serializing_if = "Option::is_none")]
45    pub meta: Option<PrivateAnalysisMeta>,
46}
47
48/// Metadata for private analysis response
49#[derive(Debug, Clone, Serialize, Deserialize)]
50pub struct PrivateAnalysisMeta {
51    #[serde(skip_serializing_if = "Option::is_none")]
52    pub file_info: Option<FileInfo>,
53}
54
55/// Reanalysis parameters
56#[derive(Debug, Clone, Serialize, Deserialize, Default)]
57pub struct ReanalyzeParams {
58    /// If true, file won't be detonated in sandbox environments
59    #[serde(skip_serializing_if = "Option::is_none")]
60    pub disable_sandbox: Option<bool>,
61
62    /// If file should have internet access in sandboxes
63    #[serde(skip_serializing_if = "Option::is_none")]
64    pub enable_internet: Option<bool>,
65
66    /// Intercept HTTPS/TLS/SSL communication
67    #[serde(skip_serializing_if = "Option::is_none")]
68    pub intercept_tls: Option<bool>,
69
70    /// Command line arguments for sandbox execution
71    #[serde(skip_serializing_if = "Option::is_none")]
72    pub command_line: Option<String>,
73
74    /// Sandbox for interactive use (defaults to "cape")
75    #[serde(skip_serializing_if = "Option::is_none")]
76    pub interaction_sandbox: Option<String>,
77
78    /// Interaction timeout in seconds (60-1800, defaults to 60)
79    #[serde(skip_serializing_if = "Option::is_none")]
80    pub interaction_timeout: Option<u32>,
81}
82
83impl ReanalyzeParams {
84    /// Create new reanalysis parameters
85    pub fn new() -> Self {
86        Self::default()
87    }
88
89    /// Set disable sandbox parameter
90    pub fn disable_sandbox(mut self, disable: bool) -> Self {
91        self.disable_sandbox = Some(disable);
92        self
93    }
94
95    /// Set enable internet parameter
96    pub fn enable_internet(mut self, enable: bool) -> Self {
97        self.enable_internet = Some(enable);
98        self
99    }
100
101    /// Set intercept TLS parameter
102    pub fn intercept_tls(mut self, intercept: bool) -> Self {
103        self.intercept_tls = Some(intercept);
104        self
105    }
106
107    /// Set command line arguments
108    pub fn command_line<S: Into<String>>(mut self, command_line: S) -> Self {
109        self.command_line = Some(command_line.into());
110        self
111    }
112
113    /// Set interaction sandbox (defaults to "cape")
114    pub fn interaction_sandbox<S: Into<String>>(mut self, sandbox: S) -> Self {
115        self.interaction_sandbox = Some(sandbox.into());
116        self
117    }
118
119    /// Set interaction timeout in seconds (60-1800, defaults to 60)
120    pub fn interaction_timeout(mut self, timeout: u32) -> Self {
121        self.interaction_timeout = Some(timeout.clamp(60, 1800));
122        self
123    }
124}
