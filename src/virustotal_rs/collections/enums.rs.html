3/// Options for collection ordering
4#[derive(Debug, Clone, Copy)]
5pub enum CollectionOrder {
6    CreationDateAsc,
7    CreationDateDesc,
8    CreationDayAsc,
9    CreationDayDesc,
10    DomainsAsc,
11    DomainsDesc,
12    FilesAsc,
13    FilesDesc,
14    IpAddressesAsc,
15    IpAddressesDesc,
16    LastModificationDateAsc,
17    LastModificationDateDesc,
18    LastModificationDayAsc,
19    LastModificationDayDesc,
20    ReferencesAsc,
21    ReferencesDesc,
22    UrlsAsc,
23    UrlsDesc,
24}
25
26impl CollectionOrder {
27    /// Convert to API parameter string
28    pub fn to_string(&self) -> &'static str {
29        match self {
30            CollectionOrder::CreationDateAsc => "creation_date+",
31            CollectionOrder::CreationDateDesc => "creation_date-",
32            CollectionOrder::CreationDayAsc => "creation_day+",
33            CollectionOrder::CreationDayDesc => "creation_day-",
34            CollectionOrder::DomainsAsc => "domains+",
35            CollectionOrder::DomainsDesc => "domains-",
36            CollectionOrder::FilesAsc => "files+",
37            CollectionOrder::FilesDesc => "files-",
38            CollectionOrder::IpAddressesAsc => "ip_addresses+",
39            CollectionOrder::IpAddressesDesc => "ip_addresses-",
40            CollectionOrder::LastModificationDateAsc => "last_modification_date+",
41            CollectionOrder::LastModificationDateDesc => "last_modification_date-",
42            CollectionOrder::LastModificationDayAsc => "last_modification_day+",
43            CollectionOrder::LastModificationDayDesc => "last_modification_day-",
44            CollectionOrder::ReferencesAsc => "references+",
45            CollectionOrder::ReferencesDesc => "references-",
46            CollectionOrder::UrlsAsc => "urls+",
47            CollectionOrder::UrlsDesc => "urls-",
48        }
49    }
50}
