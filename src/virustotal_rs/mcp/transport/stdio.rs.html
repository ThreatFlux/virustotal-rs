1use crate::mcp::server::VtMcpServer;
2use crate::mcp::McpResult;
3use crate::Client;
4use serde_json::{json, Value as JsonValue};
5use tokio::io::{AsyncBufReadExt, AsyncWriteExt, BufReader as TokioBufReader};
6
7use super::common::handle_request;
8
9/// Simple STDIO MCP server
10pub async fn run_stdio_server(client: Client) -> McpResult<()> {
11    tracing::info!("Starting `VirusTotal` MCP server with STDIO transport");
12
13    let server = VtMcpServer::new(client);
14    let stdin = tokio::io::stdin();
15    let mut stdout = tokio::io::stdout();
16    let mut reader = TokioBufReader::new(stdin);
17    let mut line = String::new();
18
19    // Send initial server info
20    let init_response = json!({
21        "jsonrpc": "2.0",
22        "result": {
23            "capabilities": server.get_server_capabilities(),
24            "serverInfo": server.get_server_info()
25        }
26    });
27    stdout
28        .write_all(format!("{}\n", init_response).as_bytes())
29        .await?;
30    stdout.flush().await?;
31
32    loop {
33        line.clear();
34        match reader.read_line(&mut line).await {
35            Ok(0) => break, // EOF
36            Ok(_) => {
37                if let Ok(request) = serde_json::from_str::<JsonValue>(line.trim()) {
38                    let response = handle_request(&server, request).await;
39                    stdout
40                        .write_all(format!("{}\n", response).as_bytes())
41                        .await?;
42                    stdout.flush().await?;
43                }
44            }
45            Err(e) => {
46                tracing::error!("Error reading from stdin: {}", e);
47                break;
48            }
49        }
50    }
51
52    Ok(())
53}
