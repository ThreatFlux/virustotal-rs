2use serde_json::{json, Value as JsonValue};

4/// Handle an MCP request
5pub async fn handle_request(server: &VtMcpServer, request: JsonValue) -> JsonValue {
6    let id = request.get("id").cloned().unwrap_or(json!(null));
7    match request.get("method").and_then(|v| v.as_str()) {
8        Some("tools/list") => tools_list(server, id).await,
9        Some("tools/call") => tools_call(server, id, request.get("params")).await,
10        Some("initialize") => initialize(server, id),
11        Some(method) => method_not_found(id, method),
12        None => invalid_request(id),
13    }
14}
15
16async fn tools_list(server: &VtMcpServer, id: JsonValue) -> JsonValue {
17    json!({
18        "jsonrpc": "2.0",
19        "id": id,
20        "result": {"tools": server.list_tools()}
21    })
22}
23
24async fn tools_call(server: &VtMcpServer, id: JsonValue, params: Option<&JsonValue>) -> JsonValue {
25    match params.and_then(|p| {
26        p.get("name")
27            .and_then(|v| v.as_str())
28            .zip(p.get("arguments"))
29    }) {
30        Some((name, args)) => match server.handle_tool_call(name, args).await {
31            Ok(result) => json!({
32                "jsonrpc": "2.0",
33                "id": id,
34                "result": {
35                    "content": [{
36                        "type": "text",
37                        "text": serde_json::to_string_pretty(&result)
38                            .unwrap_or_else(|_| result.to_string())
39                    }]}
40            }),
41            Err(error) => json!({
42                "jsonrpc": "2.0",
43                "id": id,
44                "error": {"code": -1, "message": error.to_string()}
45            }),
46        },
47        None => json!({
48            "jsonrpc": "2.0",
49            "id": id,
50            "error": {"code": -32602, "message": "Invalid params"},
51        }),
52    }
53}
54
55fn initialize(server: &VtMcpServer, id: JsonValue) -> JsonValue {
56    json!({
57        "jsonrpc": "2.0",
58        "id": id,
59        "result": {
60            "protocolVersion": "2024-11-05",
61            "capabilities": server.get_server_capabilities(),
62            "serverInfo": server.get_server_info()
63        }
64    })
65}
66
67fn method_not_found(id: JsonValue, method: &str) -> JsonValue {
68    json!({
69        "jsonrpc": "2.0",
70        "id": id,
71        "error": {
72            "code": -32601,
73            "message": format!("Method not found: {}", method),
74        }
75    })
76}
77
78fn invalid_request(id: JsonValue) -> JsonValue {
79    json!({
80        "jsonrpc": "2.0",
81        "id": id,
82        "error": {"code": -32600, "message": "Invalid request"},
83    })
