70impl<'a> IpAddressClient<'a> {
71    pub async fn get(&self, ip: &str) -> Result<IpAddress> {
72        let url = IpAddress::object_url(ip);
73        let response: ObjectResponse<IpAddressAttributes> = self.client.get(&url).await?;
74        Ok(IpAddress {
75            object: response.data,
76        })
77    }
78
79    pub async fn get_with_relationships(
80        &self,
81        ip: &str,
82        relationships: &[&str],
83    ) -> Result<IpAddress> {
84        let url = format!(
85            "{}?relationships={}",
86            IpAddress::object_url(ip),
87            relationships.join(",")
88        );
89        let response: ObjectResponse<IpAddressAttributes> = self.client.get(&url).await?;
90        Ok(IpAddress {
91            object: response.data,
92        })
93    }
94
95    pub async fn get_comments_iterator(&self, ip: &str) -> CommentIterator<'_> {
96        let url = IpAddress::relationship_objects_url(ip, "comments");
97        CommentIterator::new(self.client, url)
98    }
99
100    pub fn get_relationship_iterator<T>(
101        &self,
102        ip: &str,
103        relationship: &str,
104    ) -> CollectionIterator<'_, T>
105    where
106        T: for<'de> Deserialize<'de> + Clone,
107    {
108        let url = IpAddress::relationship_objects_url(ip, relationship);

110    }
